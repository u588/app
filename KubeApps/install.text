helm repo add bitnami https://charts.bitnami.com/bitnami
helm install --name kubeapps --namespace kubeapps bitnami/kubeapps


kubectl create serviceaccount kubeapps-operator -n kubeapps
kubectl create clusterrolebinding kubeapps-operator --clusterrole=cluster-admin --serviceaccount=kubeapps:kubeapps-operator
kubectl get clusterrolebinding.rbac.authorization.k8s.io kubeapps-operator -o yaml 
kubectl create token kubeapps-operator -n kubeapps
