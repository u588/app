kubectl apply -n kubejupter -f - <<EOF
kind: StorageClass
apiVersion: storage.k8s.io/v1
metadata:
  name: manual
provisioner: kubernetes.io/no-provisioner
volumeBindingMode: WaitForFirstConsumer
EOF


kubectl apply -n kubejupter -f - <<EOF
apiVersion: v1
kind: PersistentVolume
metadata:
  name: data-jupyter-postgresql-0
spec:
  storageClassName: manual
  capacity:
    storage: 10Gi
  accessModes:
  - ReadWriteOnce
  hostPath:
    path: /tmp
EOF
