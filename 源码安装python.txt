1、https://www.python.org/ftp/python

sudo aria2c -x8 -s8 -k 1M -c  https://www.python.org/ftp/python/3.12.7/Python-3.12.7.tgz
sudo aria2c -x8 -s8 -k 1M -c  https://www.python.org/ftp/python/3.11.9/Python-3.11.9.tgz

2、 依赖项
sudo apt update
sudo apt install build-essential zlib1g-dev libncurses5-dev libgdbm-dev libnss3-dev libssl-dev libreadline-dev libffi-dev libsqlite3-dev libbz2-dev liblzma-dev tk-dev libgdbm-compat-dev

3、解压源码包
tar xvf Python-3.x.x.tgz

4、编译和安装
cd ==> 相应目录
./configure --enable-optimization
sudo make -j $(nproc)
sudo make altinstall

5、如果需要重新编译安装，建议先执行以下命令：
sudo make clean       # 清理旧构建文件


6、使用 update-alternatives（适合多版本快速切换）

# 注册你的 Python 版本（需 root 权限，但路径可指向用户目录）
sudo update-alternatives --install /usr/local/bin/python3 python3 ~/local/python/3.10.12/bin/python3 10
sudo update-alternatives --install /usr/local/bin/python3 python3 ~/local/python/3.12.0/bin/python3 20

# 切换默认版本
sudo update-alternatives --config python3
