  daemon

  log                     global

  timeout connect         10s

  timeout server          1m

  log             global
  maxconn 10
  stats enable

  stats refresh 30s
  stats show-node
  stats show-desc Stats for openshift cluster 
  stats auth admin:openshift
  stats uri /stats
frontend openshift-api-server
    bind *:6443

    mode tcp
    option tcplog

    balance source
    mode tcp
