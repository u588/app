1. 生成 CA 根证书
openssl genrsa -out root.key 2048
openssl ecparam -genkey -name secp384r1 -out root.key
openssl req -new -x509 -days 3650 -key root.key -out root.pem

2 修改配置
cp /etc/ssl/openssl.cnf ./keys/
vim ./keys/openssl.cnf  
……
[CA_default]
# 打开copy_extensions的注释
copy_extensions = copy
……
[ req ]
# 将req_extensions的注释打开
req_extensions = v3_req # The extensions to add to a certificate request
……
[ v3_req ]
# 添加subjectAltName 
subjectAltName = @alt_names
# 文件末尾添加. www.p-pp.cn 和 *.p-pp.cn 代表允许的ServerName
[alt_names]
DNS.1 = www.tdc.io
DNS.2 = *.tdc.io

3 生成服务端证书
openssl genpkey -algorithm RSA -out server.key
openssl req -new -nodes -key server.key -out server.csr -days 3650 \
-subj "/C=cn/ST=SH/L=SH/OU=SRE/O=TST/CN=www.tdc.io" -config ./openssl.cnf -extensions v3_req

openssl x509 -req -days 3650 -in server.csr -out server.pem \
-CA root.pem -CAkey root.key -CAcreateserial -extfile ./openssl.cnf -extensions v3_req
