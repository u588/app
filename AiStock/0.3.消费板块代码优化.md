```python
# -*- coding: utf-8 -*-
# 消费板块 TSFresh 特征筛选优化系统 (2025.07)
import pandas as pd
import numpy as np 
from tsfresh import extract_features, select_features
from tsfresh.utilities.dataframe_functions import impute 
from sklearn.linear_model import LinearRegression
import yfinance as yf  # 示例数据源
 
# ======================
# 模块 1：消费数据预处理
# ======================
def preprocess_consumer_data(ticker, start_date='2024-01-01', end_date='2025-07-23'):
    """
    消费股专属数据预处理 
    重点解决：节日效应、渠道数据融合、政策哑变量
    """
    # 获取基础行情数据 (示例)
    df = yf.download(ticker, start=start_date, end=end_date)
    df = df[['Open', 'High', 'Low', 'Close', 'Volume']]
    df.columns = ['open', 'high', 'low', 'close', 'volume']
    
    # 1. 节日效应标记 (2025年消费节日日历)
    festival_dates = {
        '春节效应': ['2025-01-22', '2025-01-23', '2025-01-24'],
        '618大促': ['2025-06-18'],
        '双11预热': ['2025-10-20', '2025-10-27']
    }
    df['festival'] = 0
    for fest, dates in festival_dates.items():
        df.loc[dates, 'festival'] = 1
 
    # 2. 渠道数据融合 (示例：线上销售指数)
    # 实际应用需接入京东/天猫API (如 jd_api.get_daily_sales(ticker))
    df['online_sales_idx'] = np.random.normal(100, 15, len(df))  # 模拟数据 
 
    # 3. 消费政策哑变量 (2025年以旧换新政策)
    df['policy_2025'] = (df.index >= '2025-03-01').astype(int)
    
    # 4. 消费股特有收益率计算
    df['return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['return'] = np.where(df['return'].abs() > 0.15, 0, df['return'])  # 过滤涨跌停 
    
    return df 
 
# ======================
# 模块 2：消费特征工程 
# ======================
def extract_consumer_features(df):
    """
    消费板块定制化特征提取
    重点生成：价格弹性/品牌效应/渠道渗透特征 
    """
    # 1. 创建TSFresh所需格式
    df_reset = df.reset_index()
    df_reset['id'] = 1  # 单只股票ID设为常量 
    df_reset['sort'] = range(len(df_reset))  # 时间排序 
    
    # 2. 消费专属特征配置 (核心!)
    custom_settings = {
        # 价格弹性特征 
        'close': [
            'linear_trend_slope',  # 品牌溢价趋势
            'fft_coefficient__coeff_5__attr_"abs"',  # 节日周期 
            'binned_entropy__max_bins=5'  # 价格分布集中度
        ],
        # 渠道渗透特征 
        'online_sales_idx': [
            'mean_abs_change',  # 渠道波动性
            'c3__lag=3',  # 非线性依赖 
            'spkt_welch_density__coeff=5'  # 销售能量分布 
        ],
        # 量能特征
        'volume': [
            'cwt_coefficients__widths=(2,5,10)__coeff=7',  # 主力资金周期 
            'ar_coefficient__coeff=3__k=10'  # 量能自回归 
        ]
    }
    
    # 3. 执行特征提取
    features = extract_features(
        df_reset,
        column_id='id',
        column_sort='sort',
        default_fc_parameters=custom_settings,
        n_jobs=-1  # 使用全部CPU核心
    )
    
    # 4. 添加消费行业衍生特征
    features['premium_ratio'] = df['close'] / df['close'].rolling(30).mean()  # 品牌溢价率 
    features['channel_stability'] = df['online_sales_idx'].rolling(14).std()  # 渠道稳定性 
    
    return features 
 
# ======================
# 模块 3：四阶特征筛选
# ======================
def consumer_feature_selection(features, target):
    """
    消费特征四阶筛选法：
    1. 统计显著性 → 2. 经济逻辑 → 3. 政策稳定性 → 4. 行业中性化
    """
    # 阶段1: 统计显著性筛选 (FDR控制)
    impute(features)
    selected = select_features(features, target, fdr_level=0.001, ml_task='regression')
    
    # 阶段2: 经济逻辑验证
    valid_features = []
    for feat in selected.columns:
        # 价格弹性特征需与收益负相关 
        if 'close' in feat and np.corrcoef(selected[feat], target)[0,1] > -0.1:
            continue 
        # 渠道特征需与销量正相关 
        if 'online' in feat and np.corrcoef(selected[feat], df['online_sales_idx'])[0,1] < 0.2:
            continue 
        valid_features.append(feat)
    
    # 阶段3: 政策稳定性测试 (2025年政策断点)
    policy_date = '2025-03-01'
    pre_policy = features.loc[:policy_date, valid_features]
    post_policy = features.loc[policy_date:, valid_features]
    
    stable_features = []
    for feat in valid_features:
        pre_corr = pre_policy[feat].corr(target[:len(pre_policy)])
        post_corr = post_policy[feat].corr(target[-len(post_policy):])
        if abs(post_corr - pre_corr) < 0.15:  # 相关性变化阈值 
            stable_features.append(feat)
    
    # 阶段4: 行业中性化处理 (示例：必需消费vs可选消费)
    # 实际需接入申万行业分类数据 
    industry_dummies = pd.get_dummies(['food' if '600' in ticker else 'appliance']) 
    neutralized = LinearRegression().fit(industry_dummies, features[stable_features]).residues 
    
    return neutralized.columns.tolist()
 
# ======================
# 模块 4：实时监控系统 
# ======================
class ConsumerFeatureMonitor:
    """消费特征动态监控看板 (2025)"""
    def __init__(self, features):
        self.features = features 
        self.metrics = pd.DataFrame(index=features.columns)
        
    def update_monitoring(self, new_data):
        """月度更新监控指标"""
        # 1. 特征衰减率计算 
        current_ic = new_data.corr(target)
        decay_rate = (self.metrics['baseline_ic'] - current_ic) / self.metrics['baseline_ic']
        
        # 2. 政策敏感度检测
        policy_sensitivity = np.where(new_data.index >= '2025-06-18', 1, 0)  # 新政策标记 
        sensitivity_score = []
        for feat in self.features:
            model = LinearRegression().fit(pd.DataFrame(policy_sensitivity), new_data[feat])
            sensitivity_score.append(model.coef_[0])
        
        # 更新监控指标 
        self.metrics['last_decay'] = decay_rate
        self.metrics['policy_sensitivity'] = sensitivity_score 
        self.metrics['last_update'] = pd.Timestamp.now()
        
    def alert_rules(self):
        """2025年消费特征报警规则"""
        alerts = []
        # 规则1: 月衰减率>10%
        if any(self.metrics['last_decay'] > 0.1):
            decay_features = self.metrics[self.metrics['last_decay'] > 0.1].index.tolist()
            alerts.append(f"⚠️ 特征衰减警报: {', '.join(decay_features)}")
        
        # 规则2: 政策敏感度突变 
        if any(self.metrics['policy_sensitivity'].abs() > 0.3):
            sensitive_features = self.metrics[self.metrics['policy_sensitivity'].abs() > 0.3].index.tolist()
            alerts.append(f"🚨 政策敏感警报: {', '.join(sensitive_features)}")
        
        return alerts
 
# ======================
# 主执行流程 
# ======================
if __name__ == "__main__":
    # Step 1: 数据准备 (示例：贵州茅台)
    ticker = '600519.SS'
```
